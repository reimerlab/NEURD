[docs]    def segment_id_to_synapse_df(
        self,
        segment_id,
        *args,
        **kwargs
        ):
        
        syn_df = self.client.synapse_df_from_seg_id(
            seg_id=segment_id,
            voxel_to_nm_scaling = np.array([1,1,1])
        )

        syn_df = pu.map_column_with_dict(
            syn_df,
            "prepost",
            dict_map = dict(
                pre = "presyn",
                post = "postsyn",
            )
        )
        
        return syn_df
